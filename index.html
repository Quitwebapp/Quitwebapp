// 定义结构体类型
struct Student {
	char name[20];
	int age;
	float score;
};
int main() {
	// 初始化结构体变量
	struct Student stu1 = {"Alice", 20, 90.5};
	// 访问结构体成员
	printf("Name: %s ", stu1.name);
	printf("Age: %d ", stu1.age);
	printf("Score: %.1f ", stu1.score);
	return 0;
}
